📋 SQL — Run in Supabase SQL Editor

CREATE TABLE foods (
  id             UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title          TEXT NOT NULL,
  quantity       TEXT NOT NULL,
  location       TEXT NOT NULL,
  contact        TEXT,
  description    TEXT,
  category       TEXT DEFAULT 'other',
  listing_type   TEXT DEFAULT 'free',   -- 'free' or 'sale'
  original_price NUMERIC(10,2),         -- original MRP in ₹
  discount_pct   INTEGER,               -- e.g. 40 means 40% off
  final_price    NUMERIC(10,2),         -- discounted price
  upi_id         TEXT,                  -- seller UPI ID
  expiry_time    TIMESTAMPTZ NOT NULL,
  status         TEXT DEFAULT 'available',
  created_at     TIMESTAMPTZ DEFAULT NOW()
);

-- Allow public read/write (hackathon demo)
ALTER TABLE foods ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow all" ON foods
  FOR ALL USING (true) WITH CHECK (true);
